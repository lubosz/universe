CMAKE_MINIMUM_REQUIRED(VERSION 2.6)
project(minimal-opencl-gl4)

FIND_PACKAGE(OpenCL)
FIND_PACKAGE(OpenGL)
FIND_PACKAGE(GLEW)
FIND_PACKAGE(glm)
FIND_PACKAGE(glfw3)

INCLUDE_DIRECTORIES( 
    ${OPENGL_INCLUDE_DIR}
    ${GLEW_INCLUDE_PATH}
    ${GLFW3_INCLUDE_PATH}
    ${GLM_INCLUDE_PATH}
)

ADD_EXECUTABLE(minimal-opencl-gl4 main.cpp cll.cpp part2.cpp util.cpp)

TARGET_LINK_LIBRARIES (minimal-opencl-gl4
   ${OPENGL_LIBRARIES}
   ${GLEW_LIBRARY}
   ${OpenCL_LIBRARY}
   ${GLFW3_LIBRARY}
)
